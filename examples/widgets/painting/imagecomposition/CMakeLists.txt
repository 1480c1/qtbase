project(imagecomposition LANGUAGES CXX)

set(CMAKE_AUTORCC ON)
set(CMAKE_AUTOUIC ON)

set(INSTALL_EXAMPLEDIR "examples")

find_package(Qt5 COMPONENTS Widgets)

    main.cpp
)

    Qt::Widgets
)

    RUNTIME DESTINATION "${INSTALL_EXAMPLEDIR}"

)
